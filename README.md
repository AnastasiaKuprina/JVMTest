``` Java
package ru.netology;

public class JvmComprehension {

        public static void main(String[] args) {
            int i = 1;                      // 1 создается переменная типа int со значением 1,
                                            // выделяется и инициализируется память в стеке

            Object o = new Object();        // 2 выделяется место в куче, под класс Object
                                            // в стеке создается переменная о, которая хранит в себе ссылку на объект типа Object

            Integer ii = 2;                // 3 класс, загружающий целое число, выделяет память в куче
                                            // ссылка на этот объект памяти передается в переменную ii

            printAll(o, i, ii);             // 4 вызывается метод, появляется новый фрейм, туда передаются все три переменные


            System.out.println("finished"); // создается новый фрейм println, в него передается строка,
                                            // там создается переменная, в которую эта строка записывается,
                                            //переменная выводится на экран, фрейм закрывается
        }

        private static void printAll(Object o, int i, Integer ii) {
            Integer uselessVar = 700;    // 5 в куче создается объект Integer,
                                        // в стеке создается переменная

            System.out.println(o.toString() + i + ii);  // 6 значение в скобках вычисляются
                                                        // и передаются во фрейм Println
        }
    }
```
